h2. Install Hoccer to your project

h3. Install the .framework

* Obtain an Api Key and the framework from hoccer.com/developer.
* Add the Hoccer.framework and YAJLIOS.framework (from the lib folder) to your Xcode.
* Add both frameworks to your project. 
* Add following system frameworks:
** CoreLocation.framework
** MapKit.framework
* Add -all_load, -ObjC to Other Linker Flags in your Project Build Options.

h3. Sample Usage

h4. Using the linking service api


bc.. 
#import <Hoccer/Hoccer.h>

@interface LinkingViewController : UIViewController <HCClientDelegate> {
	HCClient *client;
}
	
@end

@implementation LinkingViewController 

- (void)viewDidLoad {
	client = [[HCClient alloc] initWithApiKey: @"1234567890" secret: @"secret!!!11elf"];
	client.delegate = self;
}

- (IBAction)send: (id)sender {
	NSDictionary *payload = [NSDictionary dictionaryWithObject: @"World" forKey: @"Hello"];
	[client send: payload withMode: HCTransferModeOneToOne];
}

- (IBAction)receive: (id)sender {
	[client receiveWithMode: HCTransferModeOneToOne];
}

#pragme mark -
#pragma mark HCClient Delegate Methods

- (void)clientDidRegister: (HCClient *)hoccer {
	NSLog(@"ready for transfering data");
}

- (void)client: (HCClient *)hoccer didSendDataWithInfo: (NSDictionary):info  {
	NSLog(@"successfully send something %@", info);
}

- (void)client: (HCClient *)client didReceiveData: (NSArray *)data {
	NSLog(@"received data: %@", data)
}

- (void)client: (HCClient *)client didFailWithError: (NSError *)error {
	NSLog(@"failed with error: %@" error);
}

- (void)clientDidUnregister: (HCClient *)hoccer {
	NSLog(@"unregistered hoccer");
}

@end
